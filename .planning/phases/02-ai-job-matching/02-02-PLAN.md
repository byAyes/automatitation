2. Main scoring function:
```typescript
export function calculateMatchScore(
  user: UserProfile,
  job: Job
): MatchScore {
  // Calculate individual scores
  const skillMatch = calculateSkillOverlap(user.skills, job.requiredSkills || []);
  const interestMatch = calculateInterestMatch(user.interests, job.category || '');
  const locationMatch = calculateLocationMatch(user.location, job.location || '', user.remoteOnly);
  const salaryMatch = calculateSalaryMatch(user.minSalary, user.maxSalary, job.salary);
  
  // Weighted average using user's weights
  const overall = 
    skillMatch * user.skillWeight +
    interestMatch * user.interestWeight +
    locationMatch * user.locationWeight +
    salaryMatch * user.salaryWeight;
  
  return {
    overall: Math.round(overall * 100) / 100,
    skillMatch: Math.round(skillMatch * 100) / 100,
    interestMatch: Math.round(interestMatch * 100) / 100,
    locationMatch: Math.round(locationMatch * 100) / 100,
    salaryMatch: Math.round(salaryMatch * 100) / 100,
    matchedSkills: [], // TODO: populate from skill matcher
    explanation: `Overall match: ${overall.toFixed(1)}% (Skills: ${skillMatch.toFixed(0)}%, Interests: ${interestMatch.toFixed(0)}%, Location: ${locationMatch.toFixed(0)}%, Salary: ${salaryMatch.toFixed(0)}%)`
  };
}
```

3. Function to score and sort jobs:
```typescript
export function scoreAndSortJobs(
  jobs: Job[],
  user: UserProfile,
  threshold: number = 70
): MatchedJob[] {
  const matchedJobs: MatchedJob[] = jobs
    .map(job => ({
      job,
      score: calculateMatchScore(user, job)
    }))
    .filter(({ score }) => score.overall >= threshold)
    .sort((a, b) => b.score.overall - a.score.overall);
  
  return matchedJobs;
}
```
